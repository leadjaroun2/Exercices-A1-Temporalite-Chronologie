    let currentIdx = 0;
    let score = 0;
    let isTrans = false;

    function loadQ() {
        isTrans = false;
        const q = questions[currentIdx];
        document.getElementById('q-text').innerText = q.q;
        document.getElementById('t-link').innerText = "👁️ Traduire en anglais / Translate to English";
        document.getElementById('q-count').innerText = `Question ${currentIdx + 1}/20`;
        document.getElementById('p-bar').style.width = ((currentIdx + 1) / 20 * 100) + "%";
        
        const optCont = document.getElementById('options');
        optCont.innerHTML = '';
        q.o.forEach((opt, i) => {
            const btn = document.createElement('button');
            btn.className = 'option-btn';
            btn.innerText = opt;
            btn.onclick = () => check(i, btn);
            optCont.appendChild(btn);
        });
        document.getElementById('feedback').style.display = 'none';
        document.getElementById('next-btn').style.display = 'none';
    }

    function toggleTranslation() {
        isTrans = !isTrans;
        document.getElementById('t-link').innerText = isTrans ? questions[currentIdx].t : "👁️ Traduire en anglais / Translate to English";
    }

    function check(i, btn) {
        const correct = questions[currentIdx].a;
        const all = document.querySelectorAll('.option-btn');
        all.forEach(b => b.disabled = true);

        const f = document.getElementById('feedback');
        f.style.display = 'block';

        if(i === correct) {
            btn.classList.add('correct');
            f.innerText = "Excellent ! Correct. ✅";
            f.style.color = "var(--success)";
            score++;
            document.getElementById('score-count').innerText = "Score: " + score;
        } else {
            btn.classList.add('incorrect');
            all[correct].classList.add('correct');
            f.innerText = "Dommage ! La réponse était : " + questions[currentIdx].o[correct] + " ❌";
            f.style.color = "var(--error)";
        }
        document.getElementById('next-btn').style.display = 'block';
    }

    function nextQuestion() {
        currentIdx++;
        if(currentIdx < 20) loadQ();
        else showRes();
    }

    function showRes() {
        document.getElementById('quiz-ui').style.display = 'none';
        document.getElementById('result-screen').style.display = 'block';
        document.getElementById('final-score').innerText = score + " / 20";
        const comm = document.getElementById('res-comment');
        if(score === 20) comm.innerHTML = "Parfait ! Vous êtes un expert de la chronologie.<br><i>Perfect! You are a chronology expert.</i>";
        else if(score >= 14) comm.innerHTML = "Très bien ! Vous comprenez bien la suite des événements.<br><i>Great! You understand the sequence of events well.</i>";
        else comm.innerHTML = "Continuez d'apprendre les connecteurs logiques !<br><i>Keep learning logical connectors!</i>";
    }

    loadQ();
